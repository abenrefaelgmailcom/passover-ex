Exercise 2: Subplot of Temperature and Rain¶

Create a subplot with two graphs in one row:¶

In [12]:
months = ["Jan", "Feb", "Mar", "Apr", "May"]
temperature = [12, 14, 17, 21, 25]
rain = [80, 60, 55, 40, 20]

plt.figure(figsize=(10, 4))

plt.subplot(1, 2, 1)
plt.plot(months, temperature, marker='o')
plt.title("Temperature by Months")
plt.xlabel("Month")
plt.ylabel("Degrees")

plt.subplot(1, 2, 2)
plt.bar(months, rain, color='blue')
plt.title("Rain by Months")
plt.xlabel("Month")
plt.ylabel("Millimeters")

plt.tight_layout()
plt.show()
